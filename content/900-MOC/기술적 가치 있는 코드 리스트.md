### 10. WMS 시스템의 실시간 데이터 동기화
**프로젝트**: `wms_core_fe`
**핵심 가치**: 물류 상태 실시간 추적
**코드 위치**: `src/hooks/useRealTimeSync.ts`
```tsx
const useRealTimeSync = (entityType: string, entityId: string) => {
  const [data, setData] = useState(null);
  
  useEffect(() => {
    const eventSource = new EventSource(`/api/stream/${entityType}/${entityId}`);
    
    eventSource.onmessage = (event) => {
      const updatedData = JSON.parse(event.data);
      setData(updatedData);
    };
    
    return () => eventSource.close();
  }, [entityType, entityId]);
  
  return data;
};
```

const useOfflineSupport = () => {
  const [isOnline, setIsOnline] = useState(navigator.onLine);
  
  useEffect(() => {
    const handleOnline = () => {
      setIsOnline(true);
      // 오프라인 중 쌓인 데이터 동기화
      syncOfflineData();
    };
    
    const handleOffline = () => {
      setIsOnline(false);
      // 캐시 모드로 전환
    };
    
    window.addEventListener('online', handleOnline);
    window.addEventListener('offline', handleOffline);
  }, []);
  
  return { isOnline };
};
